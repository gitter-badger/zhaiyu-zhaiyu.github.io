20                 {2,1,2,-5.40},

63 int max(float* begin,int total,float *max,int *idx)
64 {
65         float *p;
66         int j;
67         *max=*begin;
68         for(p=begin,j=0;p!=begin+total;++p,++j){
69                 if(*p>=*max){
70                         *max=*p;
71                         *idx=j;
72                 }
73         }
74         return 0;
75 }

84                 if(*p<=*min){
