113 function MyDiff()
114   let opt = '-a --binary '
115   if &diffopt =~ 'icase' | let opt = opt . '-i ' | endif
116   if &diffopt =~ 'iwhite' | let opt = opt . '-b ' | endif
117   let arg1 = v:fname_in
118   if arg1 =~ ' ' | let arg1 = '"' . arg1 . '"' | endif
119   let arg2 = v:fname_new
120   if arg2 =~ ' ' | let arg2 = '"' . arg2 . '"' | endif
121   let arg3 = v:fname_out
122   if arg3 =~ ' ' | let arg3 = '"' . arg3 . '"' | endif
123   let eq = ''
124   if $VIMRUNTIME =~ ' '
125     if &sh =~ '\<cmd'
126       let cmd = '""' . $VIMRUNTIME . '\diff"'
127       let eq = '"'
128     else
129       let cmd = substitute($VIMRUNTIME, ' ', '" ', '') . '\diff"'
130     endif
131   else
132     let cmd = $VIMRUNTIME . '\diff'
133   endif
134   silent execute '!' . cmd . ' ' . opt . arg1 . ' ' . arg2 . ' > ' . arg3 . eq
135 endfunction
