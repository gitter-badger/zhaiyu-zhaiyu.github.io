 6 " This color scheme uses a dark grey background.
 7 
 8 " First remove all existing highlighting.
 9 set background=dark
10 hi clear
11 if exists("syntax_on")
12   syntax reset
13 endif
14 
15 let colors_name = "evening_2"
16 
17 hi Normal ctermbg=DarkGrey ctermfg=White guifg=White guibg=grey20
18

20 hi ErrorMsg term=standout ctermbg=DarkRed ctermfg=White guibg=Red guifg=White

30 hi lCursor guibg=Cyan guifg=Black

51 " Groups for syntax highlighting
52 hi Constant term=underline ctermfg=Magenta guifg=#ffa0a0 guibg=grey5
53 hi Special term=bold ctermfg=LightRed guifg=Orange guibg=grey5
54 if &t_Co > 8
55   hi Statement term=bold cterm=bold ctermfg=Yellow guifg=#ffff60 gui=bold
56 endif
